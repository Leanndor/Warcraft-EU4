# Phlogiston
country_event = {
	id = warcraft_institution_events.100
	title = warcraft_institution_events.100.title
	desc = warcraft_institution_events.100.desc
	picture = PAINTER_ARTIST_eventPicture

	is_triggered_only = yes

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = warcraft_institution_events.100.option.a

		add_adm_power = 100
		add_dip_power = 100
		add_mil_power = 100
		add_prestige = 20

		custom_tooltip = phlogiston_penalty
		FROM = {
			add_permanent_province_modifier = {
				name = "phlogiston_origin"
				duration = -1
			}
		}
		hidden_effect = {
			every_country = {
				limit = {
					ai = no
					NOT = { tag = ROOT }
					knows_country = ROOT
				}
				country_event = { id = warcraft_institution_events.101 }
			}
			every_country = {
				limit = {
					ai = no
					NOT = { tag = ROOT }
					NOT = { knows_country = ROOT }
				}
				country_event = { id = warcraft_institution_events.102 }
			}
		}
	}
}
