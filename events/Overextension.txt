# Merchants Resented
country_event = {
	id = overextension_events.6
	title = overextension_events.6.title
	desc = overextension_events.6.desc
	picture = MERCHANTS_TALKING_eventPicture

	is_triggered_only = yes
	
	trigger = {
		num_of_merchants = 1
		any_active_trade_node = {
			NOT = { 
				has_trade_modifier = {
					who = ROOT
					name = merchant_recalled
				}
			}	
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = overextension_events.6.option.a

		random_active_trade_node = {
			limit = {
				NOT = { 
					has_trade_modifier = {
						who = ROOT
						name = merchant_recalled
					}
				}				
			}
			if = {
				limit = {
					trade_share = {
						country = ROOT
						share = 5
					}
				}
				add_trade_node_income = -10
			}
			add_trade_modifier = {
				who = root
				duration = 3650
				power = -10
				key = merchant_recalled
			}
		}
		
		if = {
			limit = { has_estate = estate_merchants }
						
			add_estate_loyalty = {
				estate = estate_merchants
				loyalty = -5
			}
		}
	}
}
