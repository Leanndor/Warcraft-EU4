namespace = rulership_events

# Random death due to disease
# [Root.Monarch.GetTitle] [Root.Monarch.GetName] falls ills
country_event = {
	id = rulership_events.1
	title = rulership_events.1.title
	desc = rulership_events.1.desc
	picture = BURNING_CASTLE_eventPicture
	
    trigger = {
        is_vunerable_to_illness = yes
        has_regency = no
    }
   
    mean_time_to_happen = {
		months = 2400
        
        modifier = {
            factor = 0.75
            NOT = { stability = 0 }
        }
    }
    
	option = {
		name = rulership_events.1.option.a
        
        kill_ruler = yes
	}
    
    option = {
		name = rulership_events.1.option.b
        
        add_years_of_income = -1
        
        random = {
            chance = 50
            kill_ruler = yes
        }
	}
}
