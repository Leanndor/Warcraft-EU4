remove_1_development_of_each = {
    if = {
        limit = {
            base_tax = 2
        }
        add_base_tax = -1
    }
    if = {
        limit = {
            base_production = 2
        }
        add_base_production = -1
    }
    if = {
        limit = {
            base_manpower = 2
        }
        add_base_manpower = -1
    }
}
